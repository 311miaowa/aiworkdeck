变更3：添加 buildFolderContext 方法

/**
 * 构建文件夹上下文：目录结构 + 文件内容摘要
 */
private String buildFolderContext(String folderId, String projectId) {
    StringBuilder sb = new StringBuilder();
    
    try {
        Long folderIdLong = Long.parseLong(folderId);
        Long projId = Long.parseLong(projectId);
        
        // 1. 获取文件夹下的所有文件
        List<ProjectFile> children = projectFileService.getFilesByParent(projId, folderIdLong);
        
        // 2. 输出目录结构
        sb.append("### Directory Structure:\n");
        for (ProjectFile f : children) {
            String type = Boolean.TRUE.equals(f.getIsFolder()) ? "[DIR]" : "[FILE]";
            sb.append("- ").append(type).append(" ").append(f.getName())
              .append(" (ID: ").append(f.getId()).append(")\n");
        }
        
        // 3. 读取非文件夹子项的内容（限制前10个）
        sb.append("\n### File Contents:\n");
        int fileCount = 0;
        for (ProjectFile f : children) {
            if (Boolean.TRUE.equals(f.getIsFolder())) continue;
            if (fileCount >= 10) {
                sb.append("\n[Truncated: Only first 10 files shown]\n");
                break;
            }
            
            String content = fileContentExtractorService.extractText(
                new java.io.File(f.getFilePath())
            );
            
            if (content != null && !content.isEmpty()) {
                sb.append("\n#### ").append(f.getName()).append(":\n");
                // 截断长内容
                if (content.length() > 20000) {
                    content = content.substring(0, 20000) + "\n[Content truncated]";
                }
                sb.append("```\n").append(content).append("\n```\n");
                fileCount++;
            }
        }
        
    } catch (Exception e) {
        sb.append("[Error reading folder content: ").append(e.getMessage()).append("]\n");
    }
    
    return sb.toString();
}

private List<AiAgentController.ContextItem> convertFileIdsToContextItems(List<String> fileIds) {
    if (fileIds == null) return null;
    return fileIds.stream().map(id -> {
        AiAgentController.ContextItem item = new AiAgentController.ContextItem();
        item.setId(id);
        item.setIsDir(false); // Legacy: assume file
        return item;
    }).collect(Collectors.toList());
}
